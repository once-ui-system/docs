The `LayoutProvider` component is a context provider that creates a listener for the Flex and Grid components to track window size.

## Simple setup

Wrap your app with `LayoutProvider`.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  codeHeight={24}
  lineNumbers
  codes={[
    {
      code:
`<LayoutProvider>
    {children}
</LayoutProvider>`,
      language: "tsx",
      label: "layout.tsx"
    }
  ]}
/>

## Advanced setup

You can pass a breakpoint object to the `LayoutProvider`. These breakpoints will only be used by the Flex and Grid components to handle runtime logic, such as `hide` and `style`. Utility classes will be applied at the standard breakpoints.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  codeHeight={32}
  lineNumbers
  codes={[
    {
      code:
`import { IconProvider, ToastProvider, ThemeProvider, DataThemeProvider, LayoutProvider } from "@once-ui-system/core";

export function Providers({ children }: {
    children: React.ReactNode
}) {
    return (
      <LayoutProvider
        breakpoints={{
          s: 640,
          m: 1024,
          l: 1200,
        }}
      >
        <ThemeProvider>
          <DataThemeProvider>
            <ToastProvider>
              <IconProvider>
                {children}
              </IconProvider>
            </ToastProvider>
          </DataThemeProvider>
        </ThemeProvider>
      </LayoutProvider>
    );
}`,
      language: "tsx",
      highlight: "1,7-13",
      label: "Providers.tsx",
    },
  ]}
/>
