A simple component for displaying important announcements, notifications, or status messages in a prominent horizontal bar. Commonly used to draw attention to critical information or updates.

<CodeBlock
  marginTop="8"
  marginBottom="16"
  codePreview={
    <Banner>
      This site uses cookies to improve your experience.
    </Banner>
  }
  codeInstances={[
    {
      code:
`<Banner>
    This site uses cookies to improve your experience.
</Banner>`,
      language: "tsx",
      label: "Banner"
    }
  ]}
/>
