A versatile component for displaying highlighted information, labels, or interactive elements in a pill-shaped container. Commonly used to draw attention to important information or provide interactive links.

  marginBottom="16"

## Interactive badges

Badges can be made interactive by adding the `href` prop, turning them into links.

<CodeBlock
  marginTop="8"
  marginBottom="16"
  codePreview={
    <Row gap="16" horizontal="center">
      <Badge title="Documentation" href="#" icon="book" />
      <Badge title="Get Started" href="#" icon="play" />
      <Badge title="View Demo" href="#" icon="eye" />
    </Row>
  }
  codeInstances={[
    {
      code:
`<Row gap="16">
    <Badge title="Documentation" href="#" icon="book" />
    <Badge title="Get Started" href="#" icon="play" />
    <Badge title="View Demo" href="#" icon="eye" />
</Row>`,
      language: "tsx",
      label: "Interactive badges"
    }
  ]}
/>

## Without arrow

Badges can be displayed without the arrow indicator by setting the `arrow` prop to `false`.

<CodeBlock
  marginTop="8"
  marginBottom="16"
  codePreview={
    <Row gap="16" horizontal="center">
      <Badge title="No Arrow" arrow={false} />
      <Badge title="With Icon" icon="info" arrow={false} />
      <Badge title="Link" href="#" arrow={false} />
    </Row>
  }
  codeInstances={[
    {
      code:
`<Row gap="16">
    <Badge title="No Arrow" arrow={false} />
    <Badge title="With Icon" icon="info" arrow={false} />
    <Badge title="Link" href="#" arrow={false} />
</Row>`,
      language: "tsx",
      label: "Badges without arrow"
    }
  ]}
/>

## Custom content

Badges can contain custom content using children instead of the `title` prop.

<CodeBlock
  marginTop="8"
  marginBottom="16"
  codePreview={
    <Row gap="16" horizontal="center">
      <Badge>
        <Flex gap="8" vertical="center">
          <Icon name="star" size="s" onBackground="brand-medium" />
          <Text onBackground="brand-strong" variant="label-strong-s">Custom Content</Text>
          <Icon name="star" size="s" onBackground="brand-medium" />
        </Flex>
      </Badge>
    </Row>
  }
  codeInstances={[
    {
      code:
`<Badge>
    <Flex gap="8" vertical="center">
      <Icon name="star" size="s" onBackground="brand-medium" />
      <Text onBackground="brand-strong" variant="label-strong-s">Custom Content</Text>
      <Icon name="star" size="s" onBackground="brand-medium" />
    </Flex>
</Badge>`,
      language: "tsx",
      label: "Custom badge content"
    }
  ]}
/>
